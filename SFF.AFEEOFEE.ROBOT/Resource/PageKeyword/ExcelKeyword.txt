OpenExcelByFile
    [Arguments]    ${FileExcel}
    Open Excel File    ${PathFile}${FileExcel}

ReadExcel
    [Arguments]    ${SheetNameExcel}    ${TitleColumn}
    ${rowcount}    Get Rows Count    ${SheetNameExcel}
    ${rowcount}    Evaluate    ${rowcount}+${1}
    @{List}    Create List
    ${Column}=    Whatscolumn_DataNewRegister    ${SheetNameExcel}    ${TitleColumn}
    : FOR    ${a}    IN RANGE    2    ${rowcount}
    \    ${data}    Get Cell Value By Sheet Name    ${SheetNameExcel}    ${Column}${a}
    \    Append To List    ${List}    ${data}
    Log    ${List}
    [Return]    @{List}

WhatsRow_DataNewRegister
    [Arguments]    ${Sheetname}    ${Title}    ${Data}
    @{DataThisColumn}=    ReadExcel    ${Sheetname}    ${Title}
    ${indexonlist}=    Get Index From List    ${DataThisColumn}    ${Data}
    ${Row}=    Evaluate    ${indexonlist} + ${2}
    [Return]    ${Row}
